    def process_payment(self, payload):
        """
        Handles payment logic.
        Priority 'Yes': Other -> Pending -> Current -> Upcoming -> Advance
        Priority 'No':  Pending -> Current -> Other -> Upcoming -> Advance
        """
        try:
            sub_id = payload.get("subscription_id")
            user_id = payload.get("user_id")
            customer_id = payload.get("customer_id")
            
            amount_paid = float(payload.get("amount") or 0)
            mode = payload.get("mode")
            pay_other_first = payload.get("pay_other_first") # Boolean

            # 1. Fetch current subscription data
            sub_res = supabase.table('subscriptions').select("*").eq("id", sub_id).single().execute()
            if not sub_res.data:
                return {"ok": False, "error": "Subscription not found"}
            
            sub = sub_res.data
            
            # 2. Get current balances
            other_charges = float(sub.get('other_service_charges') or 0)
            pending = float(sub.get('pending_amount') or 0)
            current = float(sub.get('current_amount') or 0)
            upcoming = float(sub.get('upcoming_amount') or 0)
            advance = float(sub.get('advance_balance') or 0)

            remaining = amount_paid

            # 3. EXECUTE PRIORITY (Step 1, 2, 3)
            if pay_other_first:
                # YES: Other -> Pending -> Current
                
                # 1. Other
                if remaining > 0 and other_charges > 0:
                    deduct = min(remaining, other_charges)
                    other_charges -= deduct
                    remaining -= deduct
                
                # 2. Pending
                if remaining > 0 and pending > 0:
                    deduct = min(remaining, pending)
                    pending -= deduct
                    remaining -= deduct
                
                # 3. Current
                if remaining > 0 and current > 0:
                    deduct = min(remaining, current)
                    current -= deduct
                    remaining -= deduct

            else:
                # NO: Pending -> Current -> Other
                
                # 1. Pending
                if remaining > 0 and pending > 0:
                    deduct = min(remaining, pending)
                    pending -= deduct
                    remaining -= deduct
                
                # 2. Current
                if remaining > 0 and current > 0:
                    deduct = min(remaining, current)
                    current -= deduct
                    remaining -= deduct
                
                # 3. Other
                if remaining > 0 and other_charges > 0:
                    deduct = min(remaining, other_charges)
                    other_charges -= deduct
                    remaining -= deduct

            # 4. UPCOMING (Always deducted last, before Advance)
            # This ensures money clears upcoming before becoming "Advance"
            if remaining > 0 and upcoming > 0:
                deduct = min(remaining, upcoming)
                upcoming -= deduct
                remaining -= deduct

            # 5. ADVANCE (Store whatever is left)
            if remaining > 0:
                advance += remaining

            # 6. Update Database
            update_data = {
                "other_service_charges": other_charges,
                "pending_amount": pending,
                "current_amount": current,
                "upcoming_amount": upcoming,
                "advance_balance": advance
            }
            supabase.table('subscriptions').update(update_data).eq("id", sub_id).execute()

            # 7. Record Payment
            import datetime
            payment_record = {
                "user_id": user_id,
                "customer_id": customer_id,
                "subscription_id": sub_id,
                "amount": amount_paid,
                "mode": mode,
                "date": datetime.date.today().isoformat(), # This now matches the new column
                "created_at": datetime.datetime.now().isoformat()
            }
            supabase.table('payments').insert(payment_record).execute()

            return {"ok": True}

        except Exception as e:
            return {"ok": False, "error": str(e)}

    def perform_global_expiry_check(self):
        """
        Runs immediately when App opens.
        Checks ALL subscriptions in the database (for all users).
        If today >= renewal_date, mark as 'inactive'.
        """
        try:
            import datetime
            today_str = datetime.date.today().isoformat()

            # Logic: Update ALL rows where status is 'active' AND date is today or past.
            # We use 'lte' (Less Than or Equal) to catch today's expiries too.
            
            # Note: Since RLS is disabled, this one command cleans up the entire DB.
            res = supabase.table('subscriptions')\
                .update({"status": "inactive"})\
                .eq("status", "active")\
                .lte("current_billing_end_date", today_str)\
                .execute()
                
            print(f"Global Expiry Check Ran: Updated {len(res.data)} subscriptions.")
            return {"ok": True, "count": len(res.data)}

        except Exception as e:
            print("Global Expiry Error:", e)
            return {"ok": False, "error": str(e)}

    def add_subscription_to_customer(self, payload):
        try:
            user_id = payload.get("user_id")
            cust_id = payload.get("customer_id")
            plan_id = payload.get("plan_id")
            plan_type = payload.get("type")
            plan_name = payload.get("plan_name")
            
            # --- 1. GET THE FINANCIALS (Updated) ---
            price = float(payload.get("price") or 0)
            # Fetch the new values sent from frontend
            additional = float(payload.get("additional_charge") or 0)
            discount = float(payload.get("discount_amount") or 0)

            # --- 2. CALCULATE NET AMOUNT ---
            # Formula: (Plan Price + Additional) - Discount
            final_amount = (price + additional) - discount
            if final_amount < 0: final_amount = 0

            # --- 3. Fetch Duration & Speed from Master Table ---
            duration_days = 30
            speed_val = None 

            if plan_type == 'cable':
                p_res = supabase.table('cable_plans').select('duration').eq('id', plan_id).single().execute()
                if p_res.data and p_res.data.get('duration'):
                    duration_days = int(p_res.data['duration'])
            else:
                p_res = supabase.table('internet_plans').select('duration, speed_mbps').eq('id', plan_id).single().execute()
                if p_res.data:
                    if p_res.data.get('duration'): duration_days = int(p_res.data['duration'])
                    if p_res.data.get('speed_mbps'): speed_val = str(p_res.data.get('speed_mbps')) + " Mbps"

            import datetime
            today = datetime.date.today()
            end_date = today + datetime.timedelta(days=duration_days)

            row = {
                "user_id": user_id,
                "customer_id": cust_id,
                "plan_name_cached": plan_name,
                "price": price,
                "duration": duration_days,
                "status": "active",
                "activation_date": str(today),
                "current_billing_start_date": str(today),
                "current_billing_end_date": str(end_date),
                
                # ✅ SAVE THE CALCULATED VALUES
                "current_amount": final_amount,   # Calculated Net Amount
                "upcoming_amount": final_amount,  # Starts same as current
                "additional_charge": additional,  # Save for later
                "discount_amount": discount,      # Save for later
                
                "pending_amount": 0,
                "created_at": datetime.datetime.now().isoformat()
            }

            if plan_type == 'cable':
                row['cable_plan_id'] = plan_id
            else:
                row['internet_plan_id'] = plan_id
                row['plan_mbps'] = speed_val 

            res = supabase.table('subscriptions').insert(row).execute()
            return {"ok": True} if res.data else {"ok": False, "error": "Failed to insert"}

        except Exception as e:
            return {"ok": False, "error": str(e)}

                    supabase.table('subscriptions').update({"status": "inactive"}).eq("id", sub['id']).execute()

                .order('created_at', desc=False)\

    # 3. New Function: Handle Start/Stop Logic
    def toggle_subscription_status(self, payload):
        try:
            sub_id = payload.get("subscription_id")
            action = payload.get("action") 
            
            sub_data = supabase.table('subscriptions').select("*").eq("id", sub_id).single().execute()
            if not sub_data.data:
                return {"ok": False, "error": "Subscription not found"}
            
            sub = sub_data.data

            if action == 'stop':
                supabase.table('subscriptions').update({"status": "inactive"}).eq("id", sub_id).execute()
            
            elif action == 'start':
                # 1. Get Stored Values
                stored_price = float(sub.get('price') or 0)
                stored_add = float(sub.get('additional_charge') or 0)
                stored_disc = float(sub.get('discount_amount') or 0)
                
                old_pending = float(sub.get('pending_amount') or 0)
                old_current = float(sub.get('current_amount') or 0)
                old_upcoming = float(sub.get('upcoming_amount') or 0)
                
                # ✅ FIX 1: Fetch existing Advance Balance
                advance = float(sub.get('advance_balance') or 0)

                # 2. Get Duration
                duration_days = 30
                if sub.get('cable_plan_id'):
                    p_res = supabase.table('cable_plans').select('duration').eq('id', sub['cable_plan_id']).single().execute()
                    if p_res.data and p_res.data.get('duration'): duration_days = int(p_res.data['duration'])
                elif sub.get('internet_plan_id'):
                    p_res = supabase.table('internet_plans').select('duration').eq('id', sub['internet_plan_id']).single().execute()
                    if p_res.data and p_res.data.get('duration'): duration_days = int(p_res.data['duration'])

                # 3. Shift Balances
                new_pending = old_pending + old_current

                # ✅ FIX 2: TRUST THE OLD UPCOMING VALUE
                # Even if it is 0 (Paid), we move 0 to Current. We do NOT recalculate.
                new_current = old_upcoming 

                # 4. Generate NEXT Upcoming Plan
                # (Price + Add - Disc)
                new_upcoming = (stored_price + stored_add) - stored_disc
                if new_upcoming < 0: new_upcoming = 0

                # 5. AUTO-PAY USING ADVANCE
                # If we have advance money, pay off the NEW items immediately.
                
                if advance > 0:
                    # Pay Current First (if unpaid)
                    if new_current > 0:
                        deduct = min(advance, new_current)
                        new_current -= deduct
                        advance -= deduct
                    
                    # Pay Upcoming Next
                    if advance > 0 and new_upcoming > 0:
                        deduct = min(advance, new_upcoming)
                        new_upcoming -= deduct
                        advance -= deduct

                # 6. Update Dates & Save
                import datetime
                today = datetime.date.today()
                new_renewal = today + datetime.timedelta(days=duration_days)

                update_data = {
                    "status": "active",
                    "duration": duration_days,
                    "activation_date": str(today),
                    "current_billing_end_date": str(new_renewal),
                    
                    "pending_amount": new_pending,
                    "current_amount": new_current,
                    "upcoming_amount": new_upcoming,
                    "advance_balance": advance, # ✅ Save remaining advance
                    
                    "additional_charge": stored_add,
                    "discount_amount": stored_disc,
                }
                
                supabase.table('subscriptions').update(update_data).eq("id", sub_id).execute()

            return {"ok": True}

        except Exception as e:
            return {"ok": False, "error": str(e)}

    def create_customer_with_plans(self, payload):
        try:
            customer_data = payload.get('customer')
            subscriptions_data = payload.get('subscriptions')
            user_id = customer_data.get('user_id')

            # 1. Get Customer ID
            last_cust = supabase.table('customers').select('customer_seq_id').eq('user_id', user_id).gt('customer_seq_id', 0).order('customer_seq_id', desc=True).limit(1).execute()
            next_id = 1 
            if last_cust.data and len(last_cust.data) > 0:
                current_max = last_cust.data[0]['customer_seq_id']
                if current_max is not None: next_id = current_max + 1

            # 2. Insert Customer
            cust_payload = {
                "user_id": user_id,
                "customer_seq_id": next_id,
                "name": customer_data.get('name'),
                "phone": customer_data.get('phone'),
                "alt_phone": customer_data.get('alt_phone'),
                "email": customer_data.get('email'),
                "aadhaar_number": customer_data.get('aadhaar_number'),
                "short_address": customer_data.get('short_address'), 
                "long_address": customer_data.get('long_address'),   
                "is_active": True
            }
            cust_response = supabase.table('customers').insert(cust_payload).execute()
            if not cust_response.data: return {"ok": False, "error": "Failed to create customer"}
            new_customer_id = cust_response.data[0]['id']

            # 3. Insert Plans (Fetch actual duration for each)
            import datetime
            today = datetime.date.today()
            final_subs = []

            for sub in subscriptions_data:
                # --- FETCH DURATION ---
                duration_days = 30 # Default
                plan_id = sub.get('plan_id')
                if sub.get('plan_type') == 'cable':
                    p = supabase.table('cable_plans').select('duration').eq('id', plan_id).single().execute()
                    if p.data and p.data.get('duration'): duration_days = int(p.data['duration'])
                else:
                    p = supabase.table('internet_plans').select('duration').eq('id', plan_id).single().execute()
                    if p.data and p.data.get('duration'): duration_days = int(p.data['duration'])
                
                end_date = today + datetime.timedelta(days=duration_days)

                row = {
                    "user_id": user_id,
                    "customer_id": new_customer_id,
                    "plan_name_cached": sub.get('plan_name', 'Unknown Plan'),
                    "price": sub.get('price'),
                    "duration": duration_days, # ✅ FIX: Saves real duration (e.g. 37)
                    "status": "active",
                    "activation_date": str(today),
                    "current_billing_start_date": str(today),
                    "current_billing_end_date": str(end_date),
                    "current_amount": sub.get('price'), 
                    "pending_amount": 0,
                    "upcoming_amount": sub.get('price') 
                }
                if sub.get('plan_type') == 'cable': row['cable_plan_id'] = plan_id
                else: row['internet_plan_id'] = plan_id
                final_subs.append(row)

            if final_subs: supabase.table('subscriptions').insert(final_subs).execute()
            return {"ok": True, "data": "Success"}
        except Exception as e:
            return {"ok": False, "error": str(e)}

    function calculateTotals() {
        // 1. Get Base Values

        const pend = parseFloat(document.getElementById('fPending').value) || 0;

        const discount = parseFloat(discInput.value) || 0;

    function updateRenewalDate() {
        const dateStr = document.getElementById('pLastActive').value;

        const dateObj = new Date(dateStr);
        if(isNaN(dateObj)) return;

        let dur = 30;
        const durText = document.getElementById('pDuration').value;

            const parsed = parseInt(durText);
            if (!isNaN(parsed)) dur = parsed;
        }

        dateObj.setDate(dateObj.getDate() + dur);
        
        const yyyy = dateObj.getFullYear();
        const mm = String(dateObj.getMonth() + 1).padStart(2, '0');
        const dd = String(dateObj.getDate()).padStart(2, '0');
        
        document.getElementById('pRenewal').value = `${yyyy}-${mm}-${dd}`;
